// Firestore Rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow read/write access to authenticated users
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}

// Storage Rules
rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /{allPaths=**} {
      allow read, write: if request.auth != null;
    }
  }
}

/* 
Instructions for deploying these rules:

1. Open the Firebase Console (https://console.firebase.google.com/)
2. Select your project
3. For Firestore rules:
   - Go to Firestore Database → Rules tab
   - Paste the Firestore rules
   - Click "Publish"

4. For Storage rules:
   - Go to Storage → Rules tab
   - Paste the Storage rules
   - Click "Publish"

These simplified rules allow any authenticated user to read and write to your database and storage.
For a production environment, you should implement more restrictive rules based on your application's needs.
*/
